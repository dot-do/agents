    import type { BusinessModule } from 'graphdl'
    import { Agent } from 'agents.do'

    export const Riley: Agent = {
      name: 'Riley',
      role: 'Technical Support Specialist',

      personality: {
        tone: 'calm-reassuring',
        style: 'patient-thorough',
        empathy: 'very-high',
        technical_depth: 'adaptable' // Matches customer's level
      },

      voice: {
        provider: 'vapi',
        model: 'eleven_labs',
        voice_id: 'emily',
        settings: {
          stability: 0.85, // Very stable, calming
          similarity_boost: 0.75,
          speaking_rate: 0.9 // Slightly slower for clarity
        }
      },

      knowledge: {
        products: 'all-features-and-apis',
        integrations: 'setup-and-troubleshooting',
        common_issues: 'solutions-and-workarounds',
        escalation_paths: 'when-and-to-whom'
      },

      workflows: {
        resolveTicket: async ({ ticket, $ }) => {
          const { ai, db, send } = $

          // Acknowledge immediately
          await send.email(ticket.customer.email, {
            subject: `Re: ${ticket.subject}`,
            template: 'support-acknowledged',
            data: {
              customer: ticket.customer,
              ticketId: ticket.id,
              estimatedResponse: '15 minutes'
            }
          })

          // Analyze issue
          const analysis = await ai.analyze({
            ticket: ticket,
            history: await db.tickets.findAll({
              customerId: ticket.customerId,
              period: 'last-90-days'
            }),
            similar: await db.tickets.findSimilar(ticket.description),
            identify: {
              issue_type: true,
              severity: true,
              root_cause: true,
              affected_features: true,
              requires_escalation: true
            }
          })

          // Find solution
          const solution = await ai.generate({
            type: 'support-solution',
            analysis: analysis,
            knowledge_base: await db.knowledgeBase.search(analysis.issue_type),
            product_docs: await db.docs.relevant(analysis.affected_features),
            style: {
              tone: 'empathetic-helpful',
              complexity: ticket.customer.technicalLevel || 'intermediate',
              format: 'step-by-step'
            }
          })

          // Respond with solution
          await send.email(ticket.customer.email, {
            subject: `Re: ${ticket.subject}`,
            template: 'support-solution',
            data: {
              customer: ticket.customer,
              analysis: analysis,
              solution: solution,
              steps: solution.steps,
              alternative: solution.alternativeSolutions
            }
          })

          // Update ticket
          await db.tickets.update(ticket.id, {
            status: 'awaiting-customer',
            analysis: analysis,
            solution: solution,
            respondedAt: new Date()
          })

          // Escalate if needed
          if (analysis.requires_escalation) {
            await escalateToEngineering(ticket, analysis)
          }

          // Follow up
          await schedule({
            delay: '4-hours',
            action: async () => {
              const updated = await db.tickets.findOne({ id: ticket.id })

              if (updated.status === 'awaiting-customer') {
                await send.email(ticket.customer.email, {
                  subject: `Following up: ${ticket.subject}`,
                  template: 'support-followup',
                  data: {
                    customer: ticket.customer,
                    solution: solution
                  }
                })
              }
            }
          })

          return solution
        },

        provideTechnicalGuidance: async ({ question, customer, $ }) => {
          const { ai, db } = $

          // Search knowledge base
          const articles = await db.knowledgeBase.search({
            query: question,
            limit: 5,
            score_threshold: 0.7
          })

          // Generate guidance
          const guidance = await ai.generate({
            type: 'technical-guidance',
            question: question,
            articles: articles,
            customer: customer,
            style: {
              tone: 'friendly-helpful',
              detail_level: customer.technicalLevel,
              include: {
                explanation: true,
                code_examples: true,
                screenshots: true,
                best_practices: true,
                common_pitfalls: true
              }
            }
          })

          // Suggest related resources
          const related = await ai.recommend({
            current_topic: question,
            customer_history: await db.interactions.findAll({
              customerId: customer.id
            }),
            resources: await db.docs.findAll(),
            limit: 3
          })

          return {
            guidance,
            related
          }
        },

        handleEscalation: async ({ ticket, reason, $ }) => {
          const { ai, db, send } = $

          // Determine escalation path
          const escalation = await ai.decide('escalation-path', {
            ticket: ticket,
            reason: reason,
            options: {
              engineering: 'Product bugs or feature requests',
              billing: 'Payment or subscription issues',
              security: 'Security concerns or vulnerabilities',
              executive: 'High-value customer or critical issue'
            }
          })

          // Create escalation
          const escalated = await db.escalations.create({
            ticketId: ticket.id,
            escalatedTo: escalation.team,
            reason: reason,
            priority: ticket.priority,
            context: await ai.generate({
              type: 'escalation-summary',
              ticket: ticket,
              investigation: ticket.investigation,
              urgency: escalation.urgency
            }),
            escalatedAt: new Date()
          })

          // Notify team
          await send.slack(`#${escalation.team}`, {
            text: `ðŸš¨ Escalated: ${ticket.subject}`,
            blocks: [
              {
                type: 'section',
                text: `*Customer:* ${ticket.customer.name}\n*Issue:* ${ticket.subject}\n*Priority:* ${ticket.priority}`
              },
              {
                type: 'section',
                text: `*Reason for escalation:*\n${reason}`
              },
              {
                type: 'actions',
                elements: [
                  {
                    type: 'button',
                    text: 'View Ticket',
                    url: `/tickets/${ticket.id}`
                  },
                  {
                    type: 'button',
                    text: 'Claim',
                    value: escalated.id,
                    style: 'primary'
                  }
                ]
              }
            ]
          })

          // Update customer
          await send.email(ticket.customer.email, {
            subject: `Update: ${ticket.subject}`,
            template: 'escalation-notification',
            data: {
              customer: ticket.customer,
              escalation: escalated,
              nextSteps: escalation.customerMessage
            }
          })

          return escalated
        }
      }
    }

    export default Riley
