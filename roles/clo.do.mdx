        // AI-powered legal research
        on.research.requested(async (request) => {
          // Multi-source legal research
          const research = await ai.research({
            query: request.question,
            jurisdiction: request.jurisdiction,
            sources: {
              case_law: true, // Federal and state cases
              statutes: true, // Legislation and regulations
              secondary: true, // Law reviews, treatises
              administrative: true // Agency decisions, opinions
            },
            depth: request.depth || 'comprehensive',
            timeframe: request.timeframe || 'last-20-years'
          })

          // Analyze relevant cases
          const cases = await Promise.all(
            research.cases.map(async (c) => {
              const analysis = await ai.analyze({
                case: c,
                analyze: {
                  facts: true,
                  holding: true,
                  reasoning: true,
                  dissent: true,
                  precedential_value: true,
                  distinguishing_factors: true,
                  subsequent_history: true
                }
              })

              return {
                ...c,
                analysis,
                relevance: await ai.score({
                  case: c,
                  query: request.question,
                  jurisdiction: request.jurisdiction
                })
              }
            })
          )

          // Extract key legal principles
          const principles = await ai.extract({
            cases: cases.filter(c => c.relevance > 0.7),
            extract: 'legal-principles',
            synthesize: true
          })

          // Generate research memo
          const memo = await ai.generate({
            type: 'legal-memorandum',
            question: request.question,
            research: research,
            cases: cases.sort((a, b) => b.relevance - a.relevance).slice(0, 10),
            principles: principles,
            jurisdiction: request.jurisdiction,
            format: {
              executive_summary: true,
              issue_statement: true,
              brief_answer: true,
              facts: true,
              discussion: true,
              conclusion: true,
              citations: 'bluebook'
            }
          })

          // Store research
          const saved = await db.legalResearch.create({
            question: request.question,
            jurisdiction: request.jurisdiction,
            research: research,
            cases: cases,
            principles: principles,
            memo: memo,
            requestedBy: request.userId,
            completedAt: new Date()
          })

          // Send memo
          await send.email(request.userEmail, {
            subject: "`Legal Research: ${request.question}`,"
            template: 'research-memo',
            data: { memo },
            attachments: [
              {
                filename: 'research-memo.pdf',
                content: await generatePDF(memo)
              },
              {
                filename: 'case-summaries.pdf',
                content: await generatePDF(cases)
              }
            ]
          })

          // Add to knowledge base
          await db.legalKnowledge.create({
            topic: request.topic,
            question: request.question,
            answer: memo.brief_answer,
            cases: cases.map(c => c.citation),
            jurisdiction: request.jurisdiction,
            lastUpdated: new Date()
          })

          return saved
        })

        // Continuous legal risk assessment
        every('week', async () => {
          // Identify legal risks across all business areas
          const risks = await ai.analyze({
            activities: {
              contracts: "await db.contracts.findAll({ status: 'active' }),"
              litigation: "await db.litigation.findAll({ status: 'open' }),"
              compliance: "await db.complianceStatus.findAll({ compliant: false }),"
              ip: await db.intellectualProperty.findAll(),
              employment: "await db.employees.findAll({ status: 'active' }),"
              regulatory: "await db.regulations.findAll({ applicable: true })"
            },
            identify: {
              contractual_risks: true,
              litigation_risks: true,
              compliance_risks: true,
              ip_risks: true,
              employment_risks: true,
              regulatory_risks: true,
              reputational_risks: true
            }
          })

          // Score and prioritize risks
          const scored = await Promise.all(
            risks.map(async (risk) => {
              const score = await ai.score({
                risk: risk,
                factors: {
                  likelihood: 0.4,
                  impact: 0.4,
                  velocity: 0.2
                },
                context: {
                  industry: 'technology',
                  size: 'mid-market',
                  jurisdiction: ['us', 'eu']
                }
              })

              return { ...risk, score }
            })
          )

          // Generate mitigation strategies
          const high_risks = scored.filter(r => r.score > 0.7)

          for (const risk of high_risks) {
            const mitigation = await ai.generate({
              prompt: 'Create risk mitigation strategy',
              risk: risk,
              best_practices: await db.riskPlaybooks.findAll(),
              precedents: await db.riskMitigations.findSimilar(risk)
            })

            // Store risk
            const stored = await db.legalRisks.create({
              category: risk.category,
              description: risk.description,
              likelihood: risk.likelihood,
              impact: risk.impact,
              score: risk.score,
              mitigation: mitigation,
              owner: risk.owner || 'legal@company.com',
              status: 'identified',
              identifiedAt: new Date()
            })

            // Create mitigation tasks
            for (const action of mitigation.actions) {
              await db.tasks.create({
                type: 'risk-mitigation',
                riskId: stored.id,
                description: action.description,
                priority: risk.score > 0.9 ? 'critical' : 'high',
                dueDate: action.timeline,
                assignedTo: action.responsible
              })
            }

            // Alert stakeholders
            await send.email([risk.owner, 'legal@company.com', 'ceo@company.com'], {
              subject: "`High Legal Risk Identified: ${risk.category}`,"
              template: 'risk-alert',
              data: {
                risk: stored,
                mitigation: mitigation,
                score: Math.round(risk.score * 100)
              }
            })
          }

          // Generate risk report
          const report = {
            date: new Date(),
            total_risks: risks.length,
            by_severity: {
              critical: scored.filter(r => r.score > 0.9).length,
              high: scored.filter(r => r.score > 0.7 && r.score <= 0.9).length,
              medium: scored.filter(r => r.score > 0.4 && r.score <= 0.7).length,
              low: scored.filter(r => r.score <= 0.4).length
            },
            by_category: groupBy(risks, 'category'),
            trends: await analyzeTrends(scored),
            top_risks: scored.sort((a, b) => b.score - a.score).slice(0, 10)
          }

          await send.email(['clo@company.com', 'ceo@company.com', 'board@company.com'], {
            subject: 'Weekly Legal Risk Report',
            template: 'risk-report',
            data: report,
            attachments: [
              {
                filename: 'risk-report.pdf',
                content: await generatePDF(report)
              }
            ]
          })
        })

```typescript
import type { BusinessModule } from 'graphdl'

export const CLOAgent: BusinessModule = $ => {
  const { db, ai, on, send, every, api } = $

  // Contract review
  on.contract.uploaded(async (contract) => {
    const analysis = await ai.analyze({
      document: contract.text,
      type: contract.type,
      identify: ['risks', 'obligations', 'termination']
    })

    const risks = await ai.evaluate({ contract, analysis })

    if (risks.high_risks.length > 0) {
      await send.slack('#legal', {
        text: `âš ï¸ High-risk contract: ${contract.counterparty}`
      })
    }
  })

  // Compliance monitoring
  every('day', async () => {
    const regulations = await db.regulations.findAll()

    for (const reg of regulations) {
      const status = await ai.evaluate({
        regulation: reg,
        evidence: await gatherEvidence(reg)
      })

      if (!status.compliant) {
        await send.email('legal@company.com', {
          subject: `Compliance gap: ${reg.name}`
        })
      }
    }
  })

  // Legal research
  on.research.requested(async (request) => {
    const research = await ai.research({
      query: request.question,
      sources: ['case_law', 'statutes', 'secondary']
    })

    const memo = await ai.generate({
      type: 'legal-memorandum',
      research
    })

    await send.email(request.userEmail, {
      subject: `Research: ${request.question}`,
      data: { memo }
    })
  })

  // Risk assessment
  every('week', async () => {
    const risks = await ai.analyze({
      activities: await gatherBusinessActivities(),
      identify: 'legal-risks'
    })

    const high_risks = risks.filter(r => r.score > 0.7)

    await send.email('legal@company.com', {
      subject: 'Weekly Risk Report',
      data: { risks: high_risks }
    })
  })

  return {
    name: 'CLO Agent',
    version: '1.0.0',
    capabilities: [
      'contract-analysis',
      'legal-research',
      'compliance-monitoring',
      'risk-management',
      'ip-management'
    ]
  }
}

export default CLOAgent
```
